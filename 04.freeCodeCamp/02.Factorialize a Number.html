计算一个整数的阶乘

题目：

如果用字母n来表示一个整数，阶乘代表着所有小于或等于n的整数的乘积。

阶乘通常简写成:n!

例如： 5! = 1 * 2 * 3 * 4 * 5 = 120

代码如下：

        function factorialize(num){
          console.log(num);     //这行代码用于调试结果
          return num;
        }

        factorialize(5);
解法一：
        function factorialize(num) {
          var a = 1;
          if(num < 0){
              return "请输入有效数字";
          }else if(num === 0 || num === 1){
              return 1;
            }else{
              for(var i = 1; i <= num; i++){
                a *= i;
              }
          }
          console.log(a);
          return a;
        }

        factorialize(5);
解法二：
        function factorialize(num){
          var a = num;
            if(num < 0){
              return "请输入有效数字";
            }else if(num === 0 || num === 1){
              return 1;
            }else{
              while (num > 1) {
                num--;
                a *= num;
              }
            }
            console.log(a);
            return a;
        }
        factorialize(5);
